-- Create a new folder npm-lab-1 and initialize a Node project using npm.
    mkdir npm-lab-1 
    cd npm-lab-1
    npm init 

-- Compare the difference between npm init and npm init -y
    npm init :
        Starts an interactive setup
        Asks you questions in the terminal to create package.json
    npm init -y :    
        Creates package.json instantly
        Uses default values

-- Install the package named "cors" using npm.
    npm install cors

-- Install the package named "nodemon" globally using npm
    npm install -g nodemon

-- Install the package named "express" with version "4.21.1" locally
    npm install express@4.21.1

-- Update the previously installed old version of express
    npm update express

-- Uninstall all the package that are locally installed in this lab
    npm uninstall cors express

-- Uninstall globally installed "nodemon" package
    npm uninstall -g nodemon

-- practice installing multiple packages using package.json
    npm install
    
-- write a detailed note on package .json
    package.json is the heart of a Node.js project.
    It is a metadata file that:
        Describes your project
        Manages dependencies (libraries)
        Defines scripts (commands)
        Stores configuration info
